//@version=5
indicator("Fitz FVG Engine ⚡", overlay=true, max_boxes_count=500, max_lines_count=500, max_labels_count=500)

// INPUTS
tfInput       = input.timeframe("", "FVG Timeframe (blank = chart TF)")
extendBars    = input.int(20, "Extend Boxes (bars)", minval=0)
thresholdPer  = input.float(0.0, "Threshold %", minval=0, maxval=100, step=0.1)

showBoxes     = input.bool(true,  "Show Shaded FVG Boxes")
showCE        = input.bool(true,  "Show CE (50%) Line")
ceDashed      = input.bool(true,  "CE Line Dashed")
showOpenLbl   = input.bool(true,  "Label OPEN (Unmitigated)")
showMitLbl    = input.bool(true,  "Label MITIGATED")
keepLastN     = input.int(0,      "Keep Only Last N FVGs (0 = keep all)", minval=0)

bullBoxCol    = input.color(color.new(#089981, 75), "Bullish FVG Box")
bearBoxCol    = input.color(color.new(#f23645, 75), "Bearish FVG Box")
ceBullCol     = input.color(color.new(#089981, 0),  "Bull CE Line")
ceBearCol     = input.color(color.new(#f23645, 0),  "Bear CE Line")

openBullCol   = input.color(color.new(#089981, 0),  "OPEN Bull Label")
openBearCol   = input.color(color.new(#f23645, 0),  "OPEN Bear Label")
mitBullCol    = input.color(color.new(#089981, 0),  "MIT Bull Label")
mitBearCol    = input.color(color.new(#f23645, 0),  "MIT Bear Label")

// ✅ Resolve timeframe safely (blank => chart TF)
tf = (tfInput == "" ? timeframe.period : tfInput)

// FVG DETECTION (Lux-style)
detectFVG() =>
    float thr = thresholdPer / 100.0
    bool bullSig = low > high[2] and close[1] > high[2] and ((low - high[2]) / high[2]) > thr
    bool bearSig = high < low[2] and close[1] < low[2]  and ((low[2] - high) / high) > thr
    float zTop = na
    float zBot = na
    bool  bull = na
    if bullSig
        zTop := low
        zBot := high[2]
        bull := true
    else if bearSig
        zTop := low[2]
        zBot := high
        bull := false
    [bullSig, bearSig, zTop, zBot, bull]

// single-line request.security
[bullSig, bearSig, zTop, zBot, bull] = request.security(syminfo.tickerid, tf, detectFVG(), lookahead=barmerge.lookahead_on)

// Prevent duplicates when HTF repeats signal on LTF bars
var int lastSigTime = na
bool newSig = (bullSig or bearSig) and (na(lastSigTime) or time != lastSigTime)
if newSig
    lastSigTime := time

// ARRAYS (kept perfectly aligned by always unshifting/removing together)
var float[] arrTop   = array.new_float()
var float[] arrBot   = array.new_float()
var bool[]  arrBull  = array.new_bool()
var box[]   arrBox   = array.new_box()
var line[]  arrCE    = array.new_line()
var label[] arrOpenL = array.new_label()

// ON NEW FVG
if newSig and not na(zTop) and not na(zBot)
    array.unshift(arrTop, zTop)
    array.unshift(arrBot, zBot)
    array.unshift(arrBull, bull)

    int left  = bar_index - 2
    int right = bar_index + extendBars
    float mid = (zTop + zBot) / 2.0

    box newBox = showBoxes ? box.new(left, zTop, right, zBot, border_color=na, bgcolor=(bull ? bullBoxCol : bearBoxCol)) : na
    array.unshift(arrBox, newBox)

    ceStyle = ceDashed ? line.style_dashed : line.style_solid
    line newCE = showCE ? line.new(left, mid, right, mid, xloc=xloc.bar_index, extend=extend.none, color=(bull ? ceBullCol : ceBearCol), style=ceStyle, width=1) : na
    array.unshift(arrCE, newCE)

    string openTxt = bull ? "BULL FVG (OPEN)" : "BEAR FVG (OPEN)"
    label newOpen = showOpenLbl ? label.new(bar_index, mid, openTxt, xloc=xloc.bar_index, yloc=yloc.price, style=label.style_label_left, textcolor=color.white, color=(bull ? openBullCol : openBearCol), size=size.small) : na
    array.unshift(arrOpenL, newOpen)

    // keep only last N
    if keepLastN > 0
        while array.size(arrTop) > keepLastN
            box oldBox = array.pop(arrBox)
            if not na(oldBox)
                box.delete(oldBox)

            line oldCE = array.pop(arrCE)
            if not na(oldCE)
                line.delete(oldCE)

            label oldLb = array.pop(arrOpenL)
            if not na(oldLb)
                label.delete(oldLb)

            array.pop(arrTop)
            array.pop(arrBot)
            array.pop(arrBull)

// MITIGATION CHECK
int sz = array.size(arrTop)
if sz > 0
    for i = sz - 1 to 0
        float top_i  = array.get(arrTop, i)
        float bot_i  = array.get(arrBot, i)
        bool  bull_i = array.get(arrBull, i)

        bool isMit = (bull_i and close < bot_i) or ((not bull_i) and close > top_i)

        if isMit
            box bx = array.get(arrBox, i)
            if not na(bx)
                box.delete(bx)

            line ln = array.get(arrCE, i)
            if not na(ln)
                line.delete(ln)

            label lb = array.get(arrOpenL, i)
            if not na(lb)
                label.delete(lb)

            if showMitLbl
                float mid_i = (top_i + bot_i) / 2.0
                string mitTxt = bull_i ? "BULL FVG (MITIGATED)" : "BEAR FVG (MITIGATED)"
                label.new(bar_index, mid_i, mitTxt, xloc=xloc.bar_index, yloc=yloc.price, style=label.style_label_left, textcolor=color.white, color=(bull_i ? mitBullCol : mitBearCol), size=size.small)

            array.remove(arrTop, i)
            array.remove(arrBot, i)
            array.remove(arrBull, i)
            array.remove(arrBox, i)
            array.remove(arrCE, i)
            array.remove(arrOpenL, i)

// keep CE style synced
if barstate.islast
    ceStyle2 = ceDashed ? line.style_dashed : line.style_solid
    int nCE = array.size(arrCE)
    if nCE > 0
        for j = 0 to nCE - 1
            line ln2 = array.get(arrCE, j)
            if not na(ln2)
                line.set_style(ln2, ceStyle2)
